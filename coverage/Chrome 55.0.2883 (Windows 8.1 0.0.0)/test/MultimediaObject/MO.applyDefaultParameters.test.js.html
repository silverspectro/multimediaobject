"use strict";
 
/* globals MultimediaObject config*/
var conf = config();
window._s4mConfig = {};
 
describe("MultimediaObject applyDefaultParameters", function () {
  var types = ["div", "input", "form", "img", "iframe", "video", "audio"];
 
  types.forEach(function (type) {
    var ob = new MultimediaObject(type);
    var matchAttributes = conf.defaultAttributes[type] || {};
    matchAttributes.id = "multimediaObject";
    it("should contain " + matchAttributes + " in .attributes", function () {
      expect(ob.attributes).toEqual(matchAttributes);
    });
    it("should contain " + matchAttributes + " in .element.attributes", function () {
      for (var key in matchAttributes) {
        expect(ob.element.getAttribute(key)).toEqual(matchAttributes[key]);
      }
    });
  });
 
  var ob2 = new MultimediaObject();
 
  it("should set data.autostart to true by default", function () {
    expect(ob2.data.autostart).toBe(true);
  });
  it("should set data.absoluteAssetURL to undefined", function () {
    expect(ob2.data.absoluteAssetURL).toEqual(undefined);
  });
 
  var ob3 = new MultimediaObject({
    data: {
      autostart: false
    }
  });
 
  it("should keep data.autostart to what it was set", function () {
    expect(ob3.data.autostart).toBe(false);
  });
  it("should set data.absoluteAssetURL to './'", function () {
    expect(ob3.data.absoluteAssetURL).toEqual("./");
  });
 
  var ob4 = new MultimediaObject({
    data: {
      autostart: "false",
      absoluteAssetURL: "http://test.com"
    }
  });
 
  it("should keep data.autostart to what it was set even if it's a string", function () {
    expect(ob4.data.autostart).toBe(false);
  });
  it("should set data.absoluteAssetURL to 'http://test.com'", function () {
    expect(ob4.data.absoluteAssetURL).toEqual("http://test.com");
  });
 
  window._s4mConfig.absoluteAssetURL = "http://namespace.com";
  var ob5 = new MultimediaObject({
    data: {}
  });
 
  it("should not throw an error event if autostart is not set", function () {
    expect(ob5.data.autostart).toBe(true);
  });
  it("should set data.absoluteAssetURL to 'http://namespace.com'", function () {
    expect(ob5.data.absoluteAssetURL).toEqual("http://namespace.com");
  });
 
  var ob6 = new MultimediaObject({});
 
  it("should not throw an error event if nothing i set", function () {
    expect(ob6.data.autostart).toBe(true);
  });
});
