"use strict";
 
/* global utils */
 
describe("utils.generateRandomHexColor", function () {
  it("should return a string respecting this regex /^#[a-f-0-9]{6}$/", function () {
    var maximumLoop = 100;
    var HEXRegExp = /^#[a-f-A-F-0-9]{6}$/;
    for (; maximumLoop--;) {
      expect(HEXRegExp.test(utils.generateRandomHexColor())).toBe(true);
    }
  });
});
