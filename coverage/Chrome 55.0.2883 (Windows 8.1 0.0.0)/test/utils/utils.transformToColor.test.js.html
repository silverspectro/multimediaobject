"use strict";
 
/* global utils */
 
describe("utils.transformToColor", function () {
  var rndHexColor = utils.generateRandomHexColor();
 
  var _loop = function _loop(x) {
    var colorObj = {
      r: utils.hexToR(rndHexColor),
      g: utils.hexToG(rndHexColor),
      b: utils.hexToB(rndHexColor),
      a: 1
    };
    var generatedObj = utils.transformToColor(rndHexColor);
    it("should return " + colorObj + " from " + rndHexColor, function () {
      expect(generatedObj).toEqual(colorObj);
    });
    it("object should have r propertie value between 0-255", function () {
      expect(generatedObj.r <= 255 && generatedObj.r > 0).toBeTruthy();
    });
    it("object should have g propertie value between 0-255", function () {
      expect(generatedObj.g <= 255 && generatedObj.g > 0).toBeTruthy();
    });
    it("object should have b propertie value between 0-255", function () {
      expect(generatedObj.b <= 255 && generatedObj.b > 0).toBeTruthy();
    });
  };
 
  for (var x = 0; x < 10; x++) {
    _loop(x);
  }
});
