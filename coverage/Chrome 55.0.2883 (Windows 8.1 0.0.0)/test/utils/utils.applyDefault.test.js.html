"use strict";
 
/* global utils */
 
describe("utils.applyDefaults", function () {
  var tplObj = {
    name: "test",
    hello: "yo",
    tweet: "tos"
  };
 
  var testObj = {
    test: "Hey"
  };
  var testObj2 = {
    name: "Test2"
  };
 
  it("should apply " + JSON.stringify(tplObj) + " to " + JSON.stringify(testObj), function () {
    utils.applyDefaults(testObj, tplObj);
    expect(testObj).toEqual({
      test: "Hey",
      name: "test",
      hello: "yo",
      tweet: "tos"
    });
  });
 
  it("should apply " + JSON.stringify(tplObj) + " to " + JSON.stringify(testObj2), function () {
    utils.applyDefaults(testObj2, tplObj);
    expect(testObj2).toEqual({
      name: "test",
      hello: "yo",
      tweet: "tos"
    });
  });
});
