"use strict";
 
/* global utils */
 
describe("utils.concatObject", function () {
  var obj1 = {
    test: "lol",
    test2: "test"
  };
  var obj2 = {
    test3: "test3"
  };
  var obj3 = {
    test: "lol-xptdr",
    test2: "test-test"
  };
  var obj4 = {
    test: "lol-xptdr",
    test4: "test-test"
  };
  var refObj = {
    test: "lol",
    test2: "test",
    test3: "test3"
  };
  var refObj2 = {
    test: "lol",
    test2: "test",
    test3: "test3",
    test4: "test-test"
  };
 
  it("should concatenate 2 objects", function () {
    var resultObj = utils.concatObject(obj1, obj2);
    expect(resultObj).toEqual(refObj);
  });
 
  it("should not overwrite already defined keys", function () {
    var resultObj = utils.concatObject(obj1, obj3);
    expect(resultObj).toEqual(obj1);
  });
 
  it("should concatenate multiple objects and not overwrite already defined keys", function () {
    var resultObj2 = utils.concatObject(obj1, obj2, obj3, obj4);
    expect(resultObj2).toEqual(refObj2);
  });
});
