"use strict";
 
/* global utils */
 
describe("utils.checkEvent", function () {
  var eventsList = ["click", "mousedown", "mouseup", "mousemove", "change", "touchstart", "touchmove", "touchend", "input", "focus", "dlclick", "mouseenter", "mouseleave", "mouseover", "mouseout", "blur", "search", "submit", "play", "pause", "canplay", "progress"];
  var falseList = ["test", "", " ", "myCustomEvent", 0];
 
  eventsList.forEach(function (event) {
    it("should return true " + event, function () {
      expect(utils.checkEvent(event)).toBe(true);
    });
  });
 
  falseList.forEach(function (falseEvent) {
    it("should return false for " + falseEvent, function () {
      expect(utils.checkEvent(falseEvent)).toBe(false);
    });
  });
});
