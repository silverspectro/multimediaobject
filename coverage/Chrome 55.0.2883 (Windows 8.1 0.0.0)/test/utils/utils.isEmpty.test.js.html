"use strict";
 
/* global utils */
 
describe("utils.isEmpty", function () {
  var testEmptyObject = {};
  var testFullObject = {
    accessKey: "",
    assignedSlot: null
  };
  var testCreatedNewObject = Object.create({});
 
  it("should return true if object is empty", function () {
    expect(utils.isEmpty(testEmptyObject)).toBe(true);
  });
  it("should return false if object is not empty", function () {
    expect(utils.isEmpty(testFullObject)).toBe(false);
  });
  it("should return true if object is empty and prototyped", function () {
    expect(utils.isEmpty(testCreatedNewObject)).toBe(true);
  });
});
