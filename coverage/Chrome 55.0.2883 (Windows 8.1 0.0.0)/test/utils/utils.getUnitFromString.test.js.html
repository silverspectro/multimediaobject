"use strict";
 
/* global utils */
 
describe("utils.getUnitFromString", function () {
  var units = ["%", "px", "vh", "vw", "em"];
  units.forEach(function (unit) {
    it("should return " + unit + " from '17" + unit + "'", function () {
      expect(utils.getUnitFromString("17" + unit)).toEqual(unit);
    });
  });
  it("should return px from '17tata'", function () {
    expect(utils.getUnitFromString("17tata")).toEqual("px");
  });
  it("should return px from ''", function () {
    expect(utils.getUnitFromString("")).toEqual("px");
  });
});
