'use strict';
 
/* global utils */
 
describe('utils.roundf', function () {
  var pi = Math.PI;
  var piDecimals = pi.toString().replace('3.', '');
 
  var _loop = function _loop(x) {
    var num = Number(parseFloat('3.' + piDecimals.slice(0, x)));
    it('should round the number ' + pi + ' to ' + x + ' decimals so ' + num, function () {
      expect(utils.roundf(pi, x)).toEqual(num);
    });
  };
 
  for (var x = 0; x < 15; x++) {
    _loop(x);
  }
  it('should round the number 12354.132 to 2 decimals so 12354.13', function () {
    expect(utils.roundf(12354.132, 2)).toEqual(12354.13);
  });
  it('should round the number 12354.132 to 0 decimals so 12354', function () {
    expect(utils.roundf(12354.132, 0)).toEqual(12354);
  });
});
