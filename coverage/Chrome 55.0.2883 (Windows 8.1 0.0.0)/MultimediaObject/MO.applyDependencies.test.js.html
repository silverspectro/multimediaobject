var multimediaObject = (function () {
'use strict';
 
/* globals MultimediaObject */
 
var MO_applyDependencies_test = describe('MultimediaObject applyDependencies', function () {
  var ob = new MultimediaObject();
  var dep1 = 'http://google.com';
  var dep2 = ['http://test.com', 'https://site.random.com/js'];
 
  it('should have no dependencies', function () {
    expect(ob.dependencies).toEqual([]);
  });
 
  var ob2 = new MultimediaObject();
  ob2.applyDependencies(dep1);
 
  it(("should have " + dep1 + " as a dependency"), function () {
    expect(ob2.dependencies).toEqual([dep1]);
  });
 
  var ob3 = new MultimediaObject();
  ob3.applyDependencies(dep2);
 
  it(("should have " + dep2 + " as a dependency"), function () {
    expect(ob3.dependencies).toEqual(dep2);
  });
 
  var ob4 = new MultimediaObject();
  ob4.dependencies = dep2;
  ob4.applyDependencies();
 
  it(("should have " + dep2 + " as a dependency if called with no arguments and set before as a property"), function () {
    expect(ob4.dependencies).toEqual(dep2);
  });
});
 
return MO_applyDependencies_test;
 
}());
