var multimediaObject = (function () {
'use strict';
 
/* globals MultimediaObject */
 
var MO_init_test = describe('MultimediaObject init', function () {
  var types = [
    'div',
    'input',
    'form',
    'img',
    'iframe',
    'video',
    'audio' ];
 
  types.forEach(function (type) {
    var ob = new MultimediaObject(type);
    it(("should generate a DOMElement based on the type " + type), function () {
      expect(ob.element.nodeName).toEqual(type.toUpperCase());
    });
  });
 
  var ob2 = new MultimediaObject();
 
  it('should append the element to the body', function () {
    expect(document.getElementById(ob2.name)).toEqual(ob2.element);
  });
});
 
return MO_init_test;
 
}());
