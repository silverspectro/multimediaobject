var roundf = function (v, decimal) {
  var tV = v.toString();
  var preDecimal = tV.match(/^(.+)!?\./)
        ? tV.match(/^(.+)!?\./)[0]
        : '0.';
  return Number(("" + preDecimal + (v.toString().replace(preDecimal, '').slice(0, decimal))));
};
