var closest = function (num, arr) {
  var sortedArr = arr.sort(function (a, b) { return a - b; });
  var chosen;
  for (var i = 0; i < sortedArr.length; i++) {
    var prevInd = i;
    chosen = sortedArr[i];
    if (i > 0) {
      prevInd = i - 1;
    }
    if (num < sortedArr[i]) {
      if (num - sortedArr[prevInd] < sortedArr[i] - num) {
        chosen = sortedArr[prevInd];
      } else {
        chosen = sortedArr[i];
      }
      break;
    }
  }
  return chosen;
};
